# vscode_shortcuts
ctrl + shift + ~        # create a new terminal session

# MAC

## delete
cmd + backspace     => deletes the entire line of text (no matter the app)
opt + backspace    => deletes word
## find
cmd + f             => find (in vscode)
opt + cmd + c       => match case
opt + cmd + w       => match word
opt + cmd + f       => toggles replace menu
cmd + alt + enter   => replace all
## UI/ editor
opt + z     => word wrap
f12                     # go to definition (same thing as ctrl + left-click)
shift + f12             # go to references => see where method / variable is being used
opt + f12               # peek definition => see definition of function
## terminal
ctrl + `     # focus on the terminal
cmd + 1      # focus on the coding section
cmd + shift + ]        # go to next terminal sesion
cmd + shift + [        # go to prev terminal sesion

# UBUNTU

## settings
ctrl + k + ctrl + s         => open vscode keyboard settings / shortcuts
## delete
ctrl + backspace            => deletes word
ctrl + shift + k            => deletes the entire line of text (only in vscode)
## duplicate
ctrl + shift + alt + down   => duplicates current line down (some linux distros blocks this & i have renamed to ctrl + shift + s)
ctrl + shift + alt + up     => duplicates current line up (some linux distros blocks this & i have renamed to ctrl + shift + s)
## cursor
shift + alt + down/ up      => duplicate cursor
alt + left                  => go back
alt + right                 => go forward
## find / replace
ctrl + f    => find word
alt + c     => match case
alt + w     => match word
ctrl + h    => toggles replace menu
ctrl + alt + enter  => replace all
ctrl + d                    => add selection to next find match
ctrl + shift + d            => add selection to previous find match (custom made)
## terminal
ctrl + shift + ~            # create new terminal session
ctrl + `                    # focus on the terminal
ctrl + 1                    # focus on the editor
ctrl + fn + page up         # go to next terminal sesion
ctrl + fn + page down       # go to prev terminal sesion
## UI/ editor
alt + z                     => converts word wrap to no word wrap
ctrl + __tab_num__      # choose the tab to jump to
ctrl + k + cmd + 0      # collapse all code
ctrl + k + cmd + j      # expand all code
shift + cmd + home      # collapse all folders
ctrl + -                # go back
ctrl + shift + -        # go forward
f12                     # go to definition (same thing as ctrl + left-click)
shift + f12             # go to references => see where method / variable is being used
opt + f12               # peek definition => see definition of function

# jupyter notebook
ctrl + enter    # to execute a single cell
shift + enter   # to create a new cell
d + d           # delete current cell

# on the terminal
opt + delete  # deletes the entire word (its ctrl + delete on windows)
